üìù 1. Basic Multiplication Function

A simple function with two parameters that multiplies them together:

// Basic multiplication function with parameters function multiply(a, b) { return a * b; } // Usage examples let result1 = multiply(5, 3); // Parameters: 5 and 3 let result2 = multiply(12, 4); // Parameters: 12 and 4 let result3 = multiply(7, 8); // Parameters: 7 and 8 console.log("5 √ó 3 =", result1); // Output: 15 console.log("12 √ó 4 =", result2); // Output: 48 console.log("7 √ó 8 =", result3); // Output: 56
Parameters Explained:
‚Ä¢ a - First number (multiplicand)
‚Ä¢ b - Second number (multiplier)
‚Ä¢ return - Returns the product (a √ó b)

üîß 2. Advanced Multiplication Functions

üéØ Multiple Parameters

// Function with multiple parameters function multiplyThree(a, b, c) { return a * b * c; } // Usage let result = multiplyThree(2, 3, 4); console.log("2 √ó 3 √ó 4 =", result); // 24

üîÑ Default Parameters

// Function with default parameters function multiplyWithDefault(a, b = 1) { return a * b; } // Usage console.log(multiplyWithDefault(5)); // 5 √ó 1 = 5 console.log(multiplyWithDefault(5, 3)); // 5 √ó 3 = 15

üìä Array Parameters

// Function that multiplies array elements function multiplyArray(numbers) { let result = 1; for (let num of numbers) { result *= num; } return result; } // Usage let nums = [2, 3, 4, 5]; console.log("Product:", multiplyArray(nums)); // 120

‚ö° Arrow Function

// Arrow function with parameters const multiplyArrow = (a, b) => a * b; // One-liner for multiple numbers const multiplyAll = (...numbers) => numbers.reduce((acc, num) => acc * num, 1); // Usage console.log(multiplyArrow(6, 7)); // 42 console.log(multiplyAll(2, 3, 4, 5)); // 120

‚è±Ô∏è Time Calculator

function convertToSeconds(hours, minutes, seconds = 0) { return (hours * 3600) + (minutes * 60) + seconds; } function calculateWorkPay(hours, hourlyRate, overtimeRate = 1.5) { let regularPay = Math.min(hours, 40) * hourlyRate; let overtimeHours = Math.max(0, hours - 40); let overtimePay = overtimeHours * hourlyRate * overtimeRate; return regularPay + overtimePay; }

üéØ Gaming Score

function calculateGameScore(basePoints, multiplier, bonusRound = false) { let score = basePoints * multiplier; if (bonusRound) { score = score * 2; // Double points in bonus round } return score; } // Example: 100 base points √ó 3 multiplier √ó bonus let gameScore = calculateGameScore(100, 3, true); console.log("Final Score:", gameScore); // 600

üìã 4. Function Parameter Types Summary

// 1. REQUIRED PARAMETERS - Must be provided function multiply(a, b) { return a * b; } // 2. DEFAULT PARAMETERS - Have fallback values function multiplyWithDefault(a, b = 1) { return a * b; } // 3. REST PARAMETERS - Accept unlimited arguments function multiplyAll(...numbers) { return numbers.reduce((acc, num) => acc * num, 1); } // 4. OBJECT PARAMETERS - Pass multiple values as object function calculateRectangle({length, width, height = 1}) { return length * width * height; } // 5. CALLBACK PARAMETERS - Functions as parameters function processNumbers(a, b, operation) { return operation(a, b); } // Usage examples: console.log(multiply(5, 3)); // Required: 15 console.log(multiplyWithDefault(7)); // Default: 7 console.log(multiplyAll(2, 3, 4, 5)); // Rest: 120 console.log(calculateRectangle({length: 5, width: 3})); // Object: 15 console.log(processNumbers(8, 4, multiply)); // Callback: 32
